def enforce_current_user

    render :json => ["not signed in as this user"], :status => 401
  end

def update
  @user = current_user
  if @user.update_attributes(params[:user])
    render "show_attrs"
  else
    render :json => @user.errors.full_messages, :status => 422
  end
end
